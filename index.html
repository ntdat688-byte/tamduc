const SHEET_NAME = "Trang tính 1"; // Đổi thành "Sheet1" nếu Google Sheet của bạn dùng tiếng Anh function doPost(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const doc = SpreadsheetApp.getActiveSpreadsheet(); const sheet = doc.getSheetByName(SHEET_NAME) || doc.getSheets()[0]; const data = e.parameter; // Sử dụng định dạng FormData gốc // Lấy thứ tự cột chuẩn xác từ HTML truyền lên let orderedKeys = []; if (data.orderedKeys) { orderedKeys = JSON.parse(data.orderedKeys); } else { orderedKeys = Object.keys(data); } // RẤT QUAN TRỌNG: Ép buộc vẽ lại dòng tiêu đề (Dòng 1) mỗi lần submit. // Điều này đảm bảo Tên cột và Thứ tự cột LUÔN LUÔN đúng 100% sheet.getRange(1, 1, 1, orderedKeys.length).setValues([orderedKeys]).setFontWeight("bold").setBackground("#f3f4f6"); // Đổ dữ liệu vào hàng mới theo đúng thứ tự đó const newRow = orderedKeys.map(function(key) { return data[key] === undefined ? "" : data[key]; }); sheet.appendRow(newRow); return ContentService.createTextOutput(JSON.stringify({ 'result': 'success' })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ 'error': error.message })).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } }
